import random

usa_indian_movies_languages = pd.DataFrame(columns=["imdb_title_id", "title", "year", "date_published", "genre", "language", "country", "director", "writer", "actors", "production_company", "description", "weighted_average_vote"])
for index, row in usa_india_movie_1990.iterrows():
    lan = str(row["language"]).split(", ")
    if len(lan) != 1:
        for i in range(len(lan)):
            new_row = row.copy(deep=True)
            while True:
                title_id = "tt" + str(random.randint(1000000, 2000000))
                if title_id not in usa_india_movie_1990["imdb_title_id"].to_list() and title_id not in \
                        usa_indian_movies_languages["imdb_title_id"].to_list():
                    new_row["imdb_title_id"] = title_id
                    break
            new_row["title"] = str(new_row["title"]) + f"({lan[i]})"
            new_row["language"] = lan[i]
            usa_indian_movies_languages = usa_indian_movies_languages._append(new_row)
    else:
        usa_indian_movies_languages = usa_indian_movies_languages._append(row)
